window.onload = async () => { try { await liff.init({ liffId: LIFF_ID }); if (!liff.isLoggedIn()) { liff.login(); } else { const profile = await liff.getProfile(); userId = profile.userId; // 📡 ดึงสถานะการลงทะเบียนและพิกัดจากระบบ const res = await fetch(`${API_URL}?action=getDetectedProv&uid=${userId}`); const data = await res.json(); document.getElementById('loader').style.display = 'none'; // 🔍 ตรวจสอบสิทธิ์ (The Logic) if (data.isRegistered) { // กรณีที่ 1: ลงทะเบียนแล้ว และ "ส่งรูปมาแล้ว" (มีพิกัด) -> ให้ไปหน้ากรอกงานทันที if (data.latLong || data.prov) { document.getElementById('formBody').style.display = 'block'; document.getElementById('latLong').value = data.latLong; document.getElementById('prov').value = data.prov; } else { // กรณีที่ 2: ลงทะเบียนแล้ว แต่ "เปิดเล่นๆ" (ไม่มีพิกัด) -> ปิดทึบและแจ้งเตือน document.getElementById('regForm').innerHTML = `

✅ ท่านลงทะเบียนสำเร็จแล้ว

ชื่อ: ${data.registeredName}

ระบบบันทึกชื่อท่านไว้แล้วครับ

ขั้นตอนต่อไป: ส่งภาพหน้างานในแชท LINE
เพื่อรับแบบฟอร์มบันทึกข้อมูล

`; document.getElementById('regForm').style.display = 'block'; // เพิ่ม SweetAlert เพื่อความชัดเจน (ใช้จากชุดที่ 2 ของท่าน) Swal.fire({ title: 'ยินดีต้อนรับกลับ', text: 'ท่านลงทะเบียนแล้ว ดำเนินการส่งภาพใน LINE ได้เลยครับ', icon: 'success', timer: 3000, showConfirmButton: false }); } } else { // กรณีที่ 3: ยังไม่ได้ลงทะเบียน -> แสดงฟอร์มลงทะเบียนปกติ document.getElementById('regForm').style.display = 'block'; } } } catch (err) { Swal.fire('Error', 'ไม่สามารถเชื่อมต่อฐานข้อมูลได้: ' + err.message, 'error'); } };
