window.onload = async () => { try { await liff.init({ liffId: LIFF_ID }); if (!liff.isLoggedIn()) { liff.login(); } else { const profile = await liff.getProfile(); userId = profile.userId; // 📡 ดึงข้อมูลจากหลังบ้าน const res = await fetch(`${API_URL}?action=getDetectedProv&uid=${userId}`); const data = await res.json(); document.getElementById('loader').style.display = 'none'; if (data.isRegistered) { // 🛡️ เคส: ลงทะเบียนแล้ว -> ทำการ "ปิดตาย" หน้าลงทะเบียนเดิมทันที const regForm = document.getElementById('regForm'); // ใส่ HTML เข้าไปเพื่อให้แสดงผลสวยงามและมีปุ่มที่ทำงานได้จริง regForm.innerHTML = `

✅ ลงทะเบียนแล้ว

เจ้าหน้าที่: ${data.registeredName}

ระบบบันทึกข้อมูลท่านเรียบร้อยแล้ว

ดำเนินการส่งภาพหน้างานใน LINE
เพื่อบันทึกข้อมูลงานใหม่ครับ

`; regForm.style.display = 'block'; // 🚀 เคสพิเศษ: ถ้ามีพิกัดพ่วงมาด้วย (ช่างเพิ่งส่งรูปมา) -> ข้ามไปหน้ากรอกงานทันที if (data.latLong || data.prov) { regForm.style.display = 'none'; // ซ่อนส่วนทะเบียนที่ปิดตายไว้ document.getElementById('formBody').style.display = 'block'; // โชว์ฟอร์มกรอกงาน document.getElementById('latLong').value = data.latLong; document.getElementById('prov').value = data.prov; } } else { // 🆕 เคส: ช่างใหม่ -> แสดงฟอร์มลงทะเบียนปกติ document.getElementById('regForm').style.display = 'block'; } } } catch (e) { document.getElementById('loader').innerHTML = `
Error: ${e.message}
`; } };
