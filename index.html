window.onload = async () => { try { await liff.init({ liffId: LIFF_ID }); if (!liff.isLoggedIn()) { liff.login(); } else { const profile = await liff.getProfile(); const res = await fetch(`${API_URL}?action=getDetectedProv&uid=${profile.userId}`); const data = await res.json(); document.getElementById('loader').style.display = 'none'; if (data.isRegistered) { // 🛡️ ปิดทึบหน้าลงทะเบียนทันที const regForm = document.getElementById('regForm'); regForm.innerHTML = `

✅ ลงทะเบียนแล้ว

เจ้าหน้าที่: ${data.registeredName}

ดำเนินการส่งภาพหน้างานใน LINE เพื่อบันทึกข้อมูล

`; regForm.style.display = 'block'; if (data.latLong || data.prov) { regForm.style.display = 'none'; document.getElementById('formBody').style.display = 'block'; document.getElementById('latLong').value = data.latLong; document.getElementById('prov').value = data.prov; } } else { document.getElementById('regForm').style.display = 'block'; } } } catch (e) { document.getElementById('loader').innerHTML = "Error: " + e.message; } };
