window.onload = async () => { try { await liff.init({ liffId: LIFF_ID }); if (!liff.isLoggedIn()) { liff.login(); } else { const profile = await liff.getProfile(); userId = profile.userId; const res = await fetch(`${API_URL}?action=getDetectedProv&uid=${userId}`); const data = await res.json(); document.getElementById('loader').style.display = 'none'; if (data.isRegistered) { // 🛡️ เกราะป้องกัน: ถ้าลงทะเบียนแล้วแต่ไม่มีพิกัด (เปิดเล่นๆ) ให้ไล่ไปส่งรูปใน LINE if (data.latLong || data.prov) { document.getElementById('formBody').style.display = 'block'; document.getElementById('latLong').value = data.latLong; document.getElementById('prov').value = data.prov; } else { Swal.fire({ title: 'ลงทะเบียนแล้ว', text: 'ท่านมีชื่อในระบบแล้วครับ ดำเนินการส่งภาพงานในแชท LINE ได้เลย!', icon: 'info', confirmButtonText: 'รับทราบ' }).then(() => liff.closeWindow()); } } else { document.getElementById('regForm').style.display = 'block'; } } } catch (err) { Swal.fire('Error', err.message, 'error'); } };
